/*Sowohl der return value als auch die Parameter der Funktion sind ganzzahlig in positiv.
  Wenn die Parameter vom typ unsigned char sind (8 bits),
  kann der return value maximal mit 16 bits also short dargestellt werden.*/
unsigned short kgv(unsigned char a, unsigned char b)

/*Ein Jahr hat etwas unter 10.000 Stunden daher reicht ein (unsigned)short als return value aus.
  Die Jahreszahl ist potenziell negativ (v.Chr.) aber vermutlich nicht >32.000,
  daher reicht als Parameter ein short.*/
unsigned short stunden(short jahr)

/*Der return value ist entweder 0 oder 1, daher benutzen wir den kleinsten Datentyp char.
  Die Seiten des Dreiecks sind Positiv aber nicht notwendigerweise ganzzahlig,
  daher benutzen wir unsigned float.*/
char rechtwinkel(unsigned float a, unsigned float b, unsigned float c)

/*Sowohl farenheit als auch celsius sind potentiell negativ
  aber nicht unbedingt ganzzahlig, also signed float oder double.*/
float fahrenheit(float celcius)
